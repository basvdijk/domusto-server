```
           Bas van Dijk

```

## Hardware needed
- Nefit ModuLine Easy thermostat

## Plugin configuration

```js
{
    enabled: true,
    dummyData: true,
    debug: true,
    type: 'NEFIT-EASY',
    settings: {
        minutesBetweenPolls: 60, // once per hour
        serialNumber: 'SERIALNUMBER',
        accessKey: 'ACCESSKEY',
        password: 'PASSWORD'
    }
}
```

## Device configuration

```js

// For in house temperature
{
    id: 'TEMP9',
    enabled: true,
    role: 'input',
    name: 'NEFIT in house',
    type: 'temperature',
    subType: 'temperature',
    protocol: {
        pluginId: 'NEFIT-EASY',
        id: 'inHouseTemp'
    }
},

// For outdoor temperature
{
    id: 'TEMP10',
    enabled: true,
    role: 'input',
    name: 'NEFIT outdoor',
    type: 'temperature',
    subType: 'temperature',
    protocol: {
        pluginId: 'NEFIT-EASY',
        id: 'outdoorTemp'
    }
},
```
